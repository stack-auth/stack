A component that provides a magic link sign-in form with email input and OTP verification.

## Props

This component does not accept any props.

## Example

```tsx
import { MagicLinkSignIn } from '@stackframe/stack';

export function SignInPage() {
  return (
    <div>
      <h1>Sign In</h1>
      <MagicLinkSignIn />
    </div>
  );
}
```
