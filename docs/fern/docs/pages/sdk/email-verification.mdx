# EmailVerification

A React component for handling email verification processes.

## Props

- `searchParams` (optional): `Record<string, string>` - An object containing search parameters. A valid searchParams should contain a valid `code` parameter.
- `fullPage` (optional): `boolean` - Determines whether the component should be displayed in full-page mode.

## Example

```tsx
import { EmailVerification } from '@stackframe/stack';

export function VerificationPage(props: { searchParams: Record<string, string> }) {
  return (
    <EmailVerification 
      searchParams={props.searchParams}
      fullPage={true}
    />
  );
}
```
